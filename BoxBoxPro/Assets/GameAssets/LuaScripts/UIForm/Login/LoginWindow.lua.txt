local UIBase = require "UIForm/UIBase"
local logic = LoginLogic

---@class LoginWindow : UIBase 登录界面
local LoginWindow = UIBase:New()

function LoginWindow:ManualInit()
	self.BtnPlay = self:GetButton(self.gameObject, "BtnPlay")

	self:AddListener(self.BtnPlay.onClick, self.OnLogin)
	
	self:AddEventListener(UIEventID.EventTest, self.OnCallBack)
end

function LoginWindow:OnCreate()
	UIBase:OnCreate()
	self:ManualInit()
end

function LoginWindow:OnOpen()
	UIBase:OnOpen()
end

function LoginWindow:OnClose()
	UIBase:OnClose()
end

function LoginWindow:OnDestroy()
	UIBase:OnDestroy()
	self = nil
	LoginWindow = nil
end

function LoginWindow:OnLogin()
	print("OnClickLogin")
	--CF:CloseWindow(UIFormID.LoginWindow)
	logic:Fire()
end

function LoginWindow:OnCallBack()
	print("Yeah")
end

return LoginWindow
