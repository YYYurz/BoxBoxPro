local UIBase = require "UIForm/UIBase"
local StartWindow = UIBase:New()

LuaFormManager.formClassDict["StartWindow"] = StartWindow

function StartWindow:OnCreate(transform, objs)
	UIBase.OnCreate(self, transform, objs)
	print("OnCreate")
end

function StartWindow:OnOpen(serializeID, objs)
	UIBase.OnOpen(self, serializeID, objs)
	print("OnOpen")
end

function StartWindow: OnClose()
	UIBase.OnClose(self)
	print("OnClose")
end

return StartWindow
