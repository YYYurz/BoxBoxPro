local LuaUIHelper = LuaUIHelper

---@class LuaBehaviour Lua侧的Behavior基类
local LuaBehaviour = {}

function LuaBehaviour:New(target)
	target = target or {}
	self.__index = self
	setmetatable(target, self)
	return target
end

function LuaBehaviour:GetButton(path)
	print("LuaBehaviour GetButton path: " .. path)
	return LuaUIHelper.GetButton(path)
end

return LuaBehaviour
